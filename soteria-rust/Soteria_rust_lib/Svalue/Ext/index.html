  1. | TEnum of Charon.Types.type_decl_ref

    (*

    the type decl ref of a union

    (*

    structs and tuples (ordered fields)

    *)

    (*

    arrays (all elements share the same type)

    *)

  6. | TFullPtr

and 'ghost ext_t =
  1. | Ptr of 'ghost sv * 'ghost sv option
    (*

    pointer, with optional meta

    *)
  2. | ThinPtr of 'ghost ptr
    (*

    thin pointer, without metadata but with extra info on the pointer

    *)
  3. | Enum of 'ghost sv * 'ghost sv list
    (*

    discriminant * values

    *)
  4. | Tuple of 'ghost sv list
    (*

    structs and tuples: ordered values

    *)
  5. | Array of 'ghost sv Soteria.Soteria_std.Iarray.t
    (*

    arrays: ordered values, all of the same type

    *)
  6. | Union of ('ghost sv * 'ghost sv * 'ghost sv) list
    (*

    list of blocks in the union, with their offset and size

    *)
  7. | PolyVal of Charon.Types.type_var_id
    (*

    The opaque value of a type variable, identified by (type variable index, unique identifier).

    *)
