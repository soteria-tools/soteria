Module Typed.T

type sint = [
  1. | `NonZero
  2. | `Zero
]

A symbolic integer; can either be `NonZero if it is known to not be 0, `Zero if it is 0.

type sint_ovf = [
  1. | `NonZero
  2. | `Zero
  3. | `Overflowed
]

Any symbolic integer; it may be the result of an overflowing operation

type nonzero = [
  1. | `NonZero
]

A symbolic integer known to be non-zero.

type zero = [
  1. | `Zero
]

A symbolic integer known to be zero.

type sfloat = [
  1. | `Float
]
type sbool = [
  1. | `Bool
]
type sptr = [
  1. | `Ptr
]
type sloc = [
  1. | `Loc
]
type 'a sseq = [
  1. | `List of 'a
]
type cval = [
  1. | sint
  2. | sptr
  3. | sfloat
]
type any = [
  1. | sint_ovf
  2. | sfloat
  3. | sbool
  4. | sptr
  5. | sloc
  6. | any sseq
]
