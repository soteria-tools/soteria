Module Smt_utils.RoundingMode

type t =
  1. | NearestTiesToEven
    (*

    Round to nearest, ties to even (RNE)

    *)
  2. | NearestTiesToAway
    (*

    Round to nearest, ties away from zero (RNA)

    *)
  3. | Ceil
    (*

    Round toward positive infinity (RTP)

    *)
  4. | Floor
    (*

    Round toward negative infinity (RTN)

    *)
  5. | Truncate
    (*

    Round toward zero (RTZ)

    *)
val equal : t -> t -> Ppx_deriving_runtime.bool
val pp : Ppx_deriving_runtime.Format.formatter -> t -> Ppx_deriving_runtime.unit
val show : t -> Ppx_deriving_runtime.string
val compare : t -> t -> Ppx_deriving_runtime.int
val to_sexp : t -> Simple_smt.sexp
val default : Simple_smt.sexp

Equal toquivalent to NearestTiesToAway; default mode for FloatingPointe operations.
