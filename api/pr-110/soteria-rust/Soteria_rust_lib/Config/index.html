type t = {
  1. no_compile : bool;
    (*

    Do not compile the Rust code, as it is already compiled

    *)
  2. no_timing : bool;
    (*

    Do not display execution times

    *)
  3. cleanup : bool;
    (*

    Clean up compiles files after execution

    *)
  4. ignore_leaks : bool;
    (*

    Ignore memory leaks

    *)
  5. ignore_aliasing : bool;
    (*

    Ignore pointer aliasing rules (tree borrows)

    *)
  6. monomorphize_experimental : bool;
    (*

    Use Charon's new monomorphization, which may cause unexpected results but resolves drops.

    *)
  7. with_kani : bool;
    (*

    Use the Kani library

    *)
  8. with_miri : bool;
    (*

    Use the Miri library

    *)
  9. log_compilation : bool;
    (*

    Log the compilation process

    *)
  10. step_fuel : int;
    (*

    The default step fuel for each entrypoint -- every control flow jump counts as one fuel

    *)
  11. branch_fuel : int;
    (*

    The default branch fuel for each entrypoint -- every symbolic execution branching point counts as one fuel

    *)
}
